class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super(CrossModalBridge, self).__init__()
        self.bottleneck_dim = prm.get("bottleneck_dim")
        self.num_heads = prm.get("num_heads")
        self.dropout = nn.Dropout(prm.get("dropout"))
        
        # Projection layer
        self.proj = nn.Linear(768, self.bottleneck_dim)
        self.proj_norm = nn.LayerNorm(self.bottleneck_dim)
        
        # Residual bottleneck with small transformer adapter
        self.transformer_adapter = nn.Sequential(
            nn.Linear(self.bottleneck_dim, 768),
            self.dropout,
            nn.LayerNorm(768),
            nn.ReLU(True),
            nn.Linear(768, 768),
            self.dropout,
            nn.LayerNorm(768)
        )
        
    def forward(self, visual_features, captions=None):
        x = visual_features  # [B, 197, 768]
        x = x.mean(dim=1)    # Reducing spatial dimension [B, 768]
        
        x = self.proj(x)      # [B, bottleneck_dim]
        x = self.proj_norm(x)
        x = self.dropout(x)
        
        # Apply transformer adapter
        x = x.unsqueeze(1)    # [B, 1, bottleneck_dim]
        x = self.transformer_adapter(x)  # [B, 1, 768]
        x = x.squeeze(1)      # [B, 768]
        
        # Expand to target size
        if captions is not None and captions.size(1) != x.size(0):
            x = x.unsqueeze(1).expand(-1, captions.size(1), -1)
            
        return x
