class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super(CrossModalBridge, self).__init__()
        self.bottleneck_dim = prm.get("bottleneck_dim")
        self.num_heads = prm.get("num_heads")
        self.dropout = nn.Dropout(prm.get("dropout"))
        
        # Configuration
        self.config = {
            "ndecoderlayers": 2,
            "dimension": 768,
            "nhead": self.num_heads,
            "dmodel": 768,
            "dprojection": self.bottleneck_dim,
            "dropout": self.dropout
        }
        
        # Bottleneck layers
        self.bottleneck = nn.Linear(768, self.bottleneck_dim)
        self.adapter_block = AdapterBlock(dimension=self.config["dimension"],
                                         dprojection=self.config["dprojection"],
                                         nhead=self.config["nhead"],
                                         dropout=self.config["dropout"])
        
    def forward(self, visual_features, captions=None):
        # Pass through layer norm
        x = self.dropout(visual_features)
        x = self.bottleneck(x)
        
        # Adapter block fusion
        x = self.adapter_block(x, captions=captions)
        
        return x
