class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super(CrossModalBridge, self).__init__()
        self.bottleneck_dim = prm.get("bottleneck_dim")
        self.num_heads = prm.get("num_heads")
        self.d_model = 768
        self.dropout = nn.Dropout(prm.get("dropout"))
        
        # Split into two paths for the bottleneck projection
        self.split = nn.ModuleList([
            nn.Linear(self.d_model, self.bottleneck_dim),
            nn.Linear(self.d_model, self.bottleneck_dim)
        ])
        
        # Create a transformer encoder layer
        self.transformer_layer = nn.TransformerEncoderLayer(
            d_model=self.bottleneck_dim,
            nhead=self.num_heads,
            dropout=0.1  # Default dropout from parameters
        )
        
    def forward(self, visual_features):
        # Split the visual features into two parts
        x1, x2 = torch.split(visual_features, [50, 147], dim=1)  # Split 197 into 50 and 147
        
        # Process each part through the bridge
        x1_proj = self.split[0](x1)
        x2_proj = self.split[1](x2)
        
        # Pass through the transformer layer
        x1_emb = self.transformer_layer(x1_proj)
        x2_emb = self.transformer_layer(x2_proj)
        
        # Concatenate the branches
        x_bridge = torch.cat([x1_emb, x2_emb], dim=1)
        
        # Add residual connection with dropout
        x_bridge = x_bridge + self.dropout(x_bridge)
        
        # Project back to the original embedding dimension
        x_bridge = nn.Linear(2 * self.bottleneck_dim, self.d_model)(x_bridge)
        
        return x_bridge
