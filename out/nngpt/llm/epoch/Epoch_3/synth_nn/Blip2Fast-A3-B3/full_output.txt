class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super().__init__()
        self.bottleneck_dim = prm.get('bottleneck_dim')
        self.dropout = nn.Dropout(prm.get('dropout'))
        
        # Create a small residual bottleneck
        self.encoder = nn.Sequential(
            nn.Linear(768, self.bottleneck_dim, bias=False),
            nn.ReLU(),
            nn.Linear(self.bottleneck_dim, 768),
        )
        
    def forward(self, visual_features, captions=None):
        # Residual connection
        residual = visual_features
        x = self.encoder(visual_features)
        
        # The residual is added with scaling
        x = self.dropout(x + residual)
        
        return x
