class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super().__init__()
        self.bottleneck_dim = prm.get('bottleneck_dim', 512)
        self.hidden_dim = prm.get('hidden_dim', 768)
        self.dropout = nn.Dropout(prm.get('dropout', 0.1))
        
        # Residual bottleneck layers
        self.bottleneck = nn.Linear(768, self.bottleneck_dim)
        self.visual_layer = nn.Linear(768, self.hidden_dim)
        self.visual_layer_norm = nn.LayerNorm([197, 768])
        self.visual_layer_norm_2 = nn.LayerNorm([197, 768])
        
        # Attention head
        self.att_head = MyAttentionHead(self.hidden_dim, 768)
        
        # Output layer
        self.decoder_input = nn.Linear(self.hidden_dim, 768)
    
    def forward(self, visual_features, captions=None):
        # Process through residual bottleneck
        x = self.bottleneck(visual_features)
        x = self.visual_layer_norm(x)
        # Apply self-attention
        if hasattr(self, 'att_head') and self.att_head is not None:
            x = x.transpose(1, 2)  # [B, 512, 197] → [B, 197, 512]
            x = self.att_head(x)
            x = x.transpose(1, 2)  # [B, 197, 512]
        x = self.visual_layer_norm_2(x)
        x = self.decoder_input(x)
        return x
