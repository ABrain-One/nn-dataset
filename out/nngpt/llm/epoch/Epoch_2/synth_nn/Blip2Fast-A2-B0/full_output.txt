class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super(CrossModalBridge, self).__init__()
        self.bottleneck_dim = prm.get("bottleneck_dim")
        self.num_heads = prm.get("num_heads")
        self.dropout = nn.Dropout(p=prm.get("dropout"))
        
        # Main bridge components
        self.bottleneck = nn.Linear(768, self.bottleneck_dim, bias=False)
        self.Adapter = nn.Sequential(
            nn.LayerNorm(self.bottleneck_dim),
            nn.Linear(self.bottleneck_dim, self.bottleneck_dim),
            nn.LayerNorm(self.bottleneck_dim),
            nn.Linear(self.bottleneck_dim, 768)
        )
        
        # Cross-modal transformer adapter
        self.adapter = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(
                nn.MultiheadAttention Embedding,
                self.num_heads,
                dropout=self.dropout
            ),
            num_layers=1
        )
    
    def forward(self, visual_features, captions=None):
        x = self.bottleneck(visual_features)
        x = self.Adapter(x)
        x = x + visual_features # residual connection
        x = self.dropout(x)
        
        return x
