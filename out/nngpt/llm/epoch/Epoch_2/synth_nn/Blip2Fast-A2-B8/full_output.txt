class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super(CrossModalBridge, self).__init__()
        self.config = prm
        self.bottleneck_dim = prm.get("bottleneck_dim") or 384
        self.adapter_dim = prm.get("adapter_dim") or 320
        self.num_heads = prm.get("num_heads") or 6
        
        # Creating the main cross-modal bridge structure
        # with a small transformer adapter for enhanced
        # inter-modal communication
        self.adapter_layers = nn.Sequential(
            nn.Linear(768, self.adapter_dim),
            nn.LayerNorm(self.adapter_dim),
            nn.GELU(),
            nn.Dropout(self.config.dropout),
            *[AdapterBlock(self.adapter_dim, self.adapter_dim, self.config.dropout) for _ in range(2)],
            nn.Linear(self.adapter_dim, 768)
        )

    def forward(self, visual_features, captions=None):
        x = visual_features
        # Applying an optional optional transformer block
        if self.config cru_mode and captions is not None:
            for adapter in self.adapter_layers:
                x = adapter(x, captions)
        # Applying a small transformer adapter block
        # here `cross_att` enables the bridge to encode
        # cross-modal interactions efficiently
        x = x.permute(1, 0, 2)  # [B, 197, 768] -> [197, B, 768]
        x = nn.utils.checkpoint.checkpoint(self.adapter_layers, x)
        x = x.permute(1, 0, 2)  # [197, B, 768] -> [B, 197, 768]
        
        # Projecting to obtain the final decoder embeddings
        x = self.adapter_layers[-1](x)
        # Final layer projection
        x = self.adapter_layers[-1].weight @ x + self.adapter_layers[-1].bias
        
        if self.config cru_mode and captions is not None:
            x = x.permute(1, 0, 2)  # [B, 197, 768] -> [197, B, 768]
            x = x.unsqueeze(0).permute(0, 2, 1, 3)  # [197, B, 768] -> [B, 197, 768]
        
        return x
