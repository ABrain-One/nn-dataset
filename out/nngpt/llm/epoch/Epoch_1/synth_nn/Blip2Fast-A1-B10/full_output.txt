class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super(CrossModalBridge, self).__init__()
        self.prime = prm
        
        self.drop = nn.Dropout(prm.get("dropout", 0.1))
        self.bottleneck_dim = prm.get("bottleneck_dim", 512)
        self.num_heads = prm.get("num_heads", 8)
        self.hidden_dim = prm.get("hidden_dim", 512)
        self.n_layer = prm.get("n_layer", 2)
        self.adapter_dim = prm.get("adapter_dim", 128)

        self.transformer_layers = nn.ModuleList()
        for _ in range(self.n_layer):
            self.transformer_layers.append(
                nn.TransformerEncoderLayer(
                    d_model=self.hidden_dim,
                    nhead=self.num_heads,
                    dim_feedforward=self.hidden_dim,
                    dropout=self.prime.get("dropout", 0.1),
                    activation="gelu",
                    batch_first=True,
                )
            )

        self.adapter_layers = nn.ModuleList()
        for _ in range(2):
            adapter_layer = nn.Sequential(
                nn.Linear(self.hidden_dim, self.adapter_dim),
                nn.ReLU(),
                nn.Linear(self.adapter_dim, self.hidden_dim),
            )
            self.adapter_layers.append(adapter_layer)

        self.conv_adapter = nn.Conv2d(3, 3, kernel_size=1, padding=0)
        
    def forward(self, visual_features, captions=None):
        if not captions is None:
            visual_features = visual_features.unsqueeze(1)  # [B, 1, 197, 768]
            x, _ = self.adapter_layers[0](visual_features)  # [B, 3, 197, 768] -> [B, 3, H, W]
            x = self.conv_adapter(x)  # [B, 3, H, W] -> [B, 3, H, W]
            x = x.mean(dim=1, keepdim=True)  # [B, 1, H, W]
            x = x.squeeze(dim=1)  # [B, H, W]
            x = x.permute(0, 2, 1)  # [B, W, H]
            
        x = visual_features.permute(0, 2, 1)  # [B, 768, 197]
        for layer in self.transformer_layers:
            x, _ = layer(x)
        x = x.permute(0, 2, 1)  # [B, 197, 768]
        x = self.drop(x)
        
        visual_embeds = x
