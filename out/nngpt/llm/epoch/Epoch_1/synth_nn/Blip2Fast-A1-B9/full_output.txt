class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super(CrossModalBridge, self).__init__()
        self.bottleneck_dim = prm.get("bottleneck_dim")
        self.num_heads = prm.get("num_heads")
        self.dropout = prm.get("dropout")
        
        # Define layers
        self.slf_attn = nn.MultiheadAttention(self.bottleneck_dim, self.num_heads)
        selfproj_in = nn.Linear(768, self.bottleneck_dim, bias=False)
        selfnorm_in = nn.LayerNorm(self.bottleneck_dim)
        selfproj_after_attn = nn.Linear(self.bottleneck_dim, 768, bias=False)
        selfnorm_after = nn.LayerNorm(768)
        
    def forward(self, visual_features, captions=None):
        x = visual_features
        B = x.size(0)
        
        # Project and normalize input features
        x = selfproj_in(x)
        x = selfnorm_in(x)
        
        # Apply self-attention with residual connection
        x = x + self.slf_attn(x)[0]
        x = F.dropout(x, p=self.dropout, training=self.training)
        x = selfnorm_after(x)
        
        # Project back to original dimension
