class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super().__init__()
        self.bottleneck_dim = prm.get("bottleneck_dim")
        self.num_heads = prm.get("num_heads")
        self.dropout = prm.get("dropout")
        
        self.slf_attn = nn.MultiheadAttention(self.bottleneck_dim, self.num_heads, dropout=self.dropout)
        self.proj_in = nn.Linear(768, self.bottleneck_dim)
        self.proj_out = nn.Linear(self.bottleneck_dim, 768)
        self.dropout_layer = nn.Dropout(self.dropout)
        selflayernorm = nn.LayerNorm(768)
    
    def forward(self, visual_features, captions=None):
        x = self.proj_in(visual_features)
        x, _ = self.slf_attn(x)
        x = self.proj_out(x)
        x = selflayernorm(visual_features + x)
        x = self.dropout_layer(x)
        return x
