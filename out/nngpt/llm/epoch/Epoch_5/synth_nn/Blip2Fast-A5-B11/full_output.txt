class CrossModalBridge(nn.Module):
    def __init__(self, prm):
        super().__init__()
        self.bottleneck_dim = prm.get("bottleneck_dim")
        self.num_heads = prm.get("num_heads")
        self.dropout = nn.Dropout(p=prm.get("dropout"))
        
        # Configuring the decoder-compatible transformer block
        dim = 768
        self.slf_attn = nn.MultiheadAttention(
            embed_dim=dim, 
            num_heads=self.num_heads, 
            dropout=self.dropout
        )
        
        # Residual bottleneck with layer norm
        self.bottleneck = nn.Sequential(
            nn.Linear(dim, self.bottleneck_dim, bias=False),
            nn.ReLU(),
            nn.Linear(self.bottleneck_dim, dim)
        )
        
        # Final output layer
        self.proj = nn.Linear(dim, dim)
        
    def forward(self, visual_features, captions=None):
        x = visual_features
        
        # Applying self-attention
        x, _ = self.slf_attn(x, x, x)
        x = x + self.dropout(x)  # Adding residual connection
        
        # Applying bottleneck
        x = self.bottleneck(x)
        x = self.proj(x)
        
        # Final output with scaled init
        return x * (1/(self.bottleneck_dim ** 0.5))
