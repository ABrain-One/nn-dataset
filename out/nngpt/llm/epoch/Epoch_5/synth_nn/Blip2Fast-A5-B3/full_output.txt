            out_features=768,
            bias=True,
        )
        self.slf_dec = nn.Linear(
            in_features=768,
            out_features=768,
            bias=True,
        )
        self.slf_norm = nn.LayerNorm(768)
        
    def forward(self, visual_features, captions=None):
