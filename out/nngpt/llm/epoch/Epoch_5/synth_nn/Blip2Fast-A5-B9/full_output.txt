        # Self-attention layers

                d_model=768,
                nhead=prm.get('num_heads'),

            ),
        )

        self.res_proj = nn.Linear(768, self.reducer_dim, bias=False)
        self.res_MHA = nn.MultiheadAttention(self.reducer_dim, 3)
